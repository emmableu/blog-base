## solution: 双栈
![](https://raw.githubusercontent.com/emmableu/image/master/224-1.png)
```python
class Solution:
    def calculate(self, s):
        s = "(" + s.replace(" ", "").replace("(-", "(0-") + ")"
        ops, nums = [], []
        i = 0
        while i < len(s):
            c = s[i]
            i += 1
            if c.isdigit():
                while i < len(s) and s[i].isdigit():
                    c += s[i]
                    i += 1
                nums.append(int(c))
            elif c == "(":
                ops.append(c)
            elif c == ")":
                while ops and ops[-1] != "(":
                    self.calc(nums, ops)
                ops.pop() # 全都计算完后，pop出左括号
            elif c in "+-":
                # 要把ops里面有的,在（后面的先算了
                while ops and ops[-1] != "(":
                    self.calc(nums, ops)
                ops.append(c)
        return nums[0]


    def calc(self, nums, ops):
        op, y, x = ops.pop(), nums.pop(), nums.pop() if nums else 0
        if op == "+":
            res = x + y
        elif op == "-":
            res = x - y
        nums.append(res)
```
