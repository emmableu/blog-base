具体解释见 backtracking 模版

```python
import copy
class Solution:
	def permute(self, nums):
		res = []
		used = [False] * len(nums)
		self.dfs(nums, [], used, res)
		return res

	def dfs(self, nums, path, used, res):
		if len(path) == len(nums):
			res.append(copy.deepcopy(path))
			return
		for i in range(len(nums)):
			if not used[i]:
				path.append(nums[i])
				used[i] = True
				# 进入下一层决策树
				self.dfs(nums, path, used, res)
				# 取消选择
				path.pop()
				used[i] = False
```
